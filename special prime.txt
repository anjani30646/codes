special prime(codechef)

1 . It is a prime.
2. It has exactly two distinct multiples in the given array.

Find the smallest special prime in the sequence. If no element of array satisfies the above then print âˆ’1.

Sample Input0:
5
2 2 3 4 5


Sample Output:
2

EXPLANATION:
In the given case 2 is the only integer that has exactly 2 unique multiples which are 2,4


#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main() 
{
    int n;
    cin >> n;
    int a[n],i,j,k;
    for(i=0;i<n;i++)
    {
        cin >> a[i];
    }
    sort(a,a+n);
    int flag=0;
    for(i=0;i<n;i++)
    {
        int c=0;
        for(j=2;j<=a[i]/2;j++)
        {
            if(a[i]%j==0)
                c++;
            if(c>1)
                break;
        }
        int count=0;
        if(c==0 || a[i]==2)
        {
            for(k=i+1;k<=n;k++)
            {
                if(a[i]!=a[k] && a[k]%a[i]==0)
                    count++;
            }
        }
        if(count==1)
            {
                flag=1;
                cout << a[i];
                break;
            }
    }
    if(flag==0)
        cout << -1;
	return 0;
}
