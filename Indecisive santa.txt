#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() 
{
    int m,n;
    cin >> m >>n;
    int x1,y1,x2,y2;
    cin >>x1 >>y1 >>x2 >>y2;
    int i,j,d1,d2,c=0;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
        {
            d1=abs(i-x1)+abs(j-y1);
            d2=abs(i-x2)+abs(j-y2);
            if(d1==d2)
            {
                //cout << i << " " << j << endl;
                 c++;   
            }
        }
    }
    cout << c;
    return 0;
}
